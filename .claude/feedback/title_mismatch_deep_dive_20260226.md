```python
# src/post_processing/text_cleaner.py
from opencc import OpenCC

converter = OpenCC('s2tw')  # 简体→繁体（台湾标准）

def clean_ocr_text(text: str) -> str:
    """清洗 OCR 文本"""
    if not text:
        return text

    # 1. 繁简转换
    text = converter.convert(text)

    # 2. 形似字修正（书名上下文）
    text = text.replace('圃', '圖')  # 圃→圖

    # 3. 噪声清理
    text = text.replace('来緯', '緯')  # 来緯→緯

    return text
```
